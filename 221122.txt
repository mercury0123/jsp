=================================domain패키지 아래 CarVo.java

package domain;

public class CarVO {
	//멤버변수 String num, String name, int year 선언
	private String num;
	private String name;
	private int year;
	
	//생성자
	public CarVO(String num, String name, int year) {
		this.num = num;
		this.name = name;
		this.year = year;
	}

	//getter setter
	public String getNum() {
		return num;
	}

	public void setNum(String num) {
		this.num = num;
	}

	public String getName() {
		return name;
	}

	public void setName(String name) {
		this.name = name;
	}

	public int getYear() {
		return year;
	}

	public void setYear(int year) {
		this.year = year;
	}
	
}

<%@ page language="java" contentType="text/html; charset=UTF-8"
    pageEncoding="UTF-8"%>
    
<%@ taglib uri="http://java.sun.com/jsp/jstl/core" prefix="c" %>
<!DOCTYPE html>
<html>
<head>
<meta charset="UTF-8">
<title>choose</title>
</head>
<body>
이름 : ${param.userName}<br>
나이 : ${param.userAge}<br>
<!-- xx님은 ㅇㅇ세 입니다 -->
<!-- choose : 다중조건 -->
<!-- 

<%@ taglib uri="http://java.sun.com/jsp/jstl/core" prefix="c" %> 이거 찍어야 아래 명령문 실행됨 

<c:choose> 
	<c:when test="조건">
		실행문
	</c:when>	
	<c:when test="조건">
		실행문
	</c:when>	
	<c:otherwise>
		나머지 else 
	</c:otherwise>
</c:choose> 
-->
<!-- 
19세 이상이면 xx님은 ㅇㅇ세 입니다. 성인입니다 같이 출력 
13세 이상이면 xx님은 ㅇㅇ세 입니다. 청소년입니다
1~13세 이면  xx님은 ㅇㅇ세 입니다. 어린이입니다
나머지는 아직 태어나지 않았습니다
-->
<c:choose> 
	<c:when test="${param.userAge >= 19}">
		${param.userName}님은  ${param.userAge}세, '성인'입니다 <br>
	</c:when>
	<c:when test="${param.userAge >= 13}">
		${param.userName}님은  ${param.userAge}세, '청소년'입니다 <br>
	</c:when>
	<c:when test="${param.userAge >= 1}">
		${param.userName}님은  ${param.userAge}세, '어린이'입니다 <br>
	</c:when>
	<c:otherwise>
		아직 태어나지 않았습니다.<br>
	</c:otherwise>
</c:choose>
</body>
</html>
